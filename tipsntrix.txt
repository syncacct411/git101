###################### GIT COMMANDS ########################

git diff commit1 commit2	# COMPARE TWO COMMITS
git diff 			# COMPARE WORKING-DIRECTORY AGAINST STAGING-AREA
git diff --staged		# COMPARE THE STAGING AGAINST THE REPOSITORY 

git log --graph --oneline master experimental 	# SHOW CHANGES BETWEEN "master" AND "experimental" ON ONE LINE

git show commit2	# SHOW CHANGES THAT WERE MADE/INTRODUCED COMPARED TO ITS PARENT

git merge master experimental	# MERGE BRANCHES; BESURE YOU'RE ON "master" BRANCH

git branch -d experimental	# REMOVE/DELETE A BRANCH LABEL, BUT KEEPING ALL COMMITS ON THAT BRANCH


#SYNTAX FOR CREATING A BRANCH ON A REMOTE(e.g., GitHub) LOCATION
git remote add branch_name url_for_the_remote

git remote add origin https://github.com/syncacct411/git101.git	#CREATING A REMOTE BRANCH CALLED "origin"

git push to_remote_location branch_selection_is_from_local_master_branch
git push origin master

git pull from_remote_location branch_selection_is_from_remote_master_branch_on_github
git pull origin master

#NOTE: 
git pull origin master =	git fetch origin	# UPDATE THE LOCAL VERSION WITH THE REMOTE BRANCH CALLED "origin" LEAVING THE LOCAL VERSION ALONE
